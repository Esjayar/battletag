

VAR nb_hit		SEND
VAR nb_shot		SEND


STATE init 

FIRST_STATE

EVENT ENTER_STATE
HUD_DIGIT_OFF	
HUD_ICON_ON	BULLET
HUD_JAUGE_BLINK	1
SET_HARNESS	1
SET	nb_hit 	0
SET	nb_shot	0
SND	START
GOTO game






















































END_EVENT

END_STATE

STATE game 

EVENT BUTTON_2_JUST_PRESSED
ANIM		ASHT
INC		nb_shot
SND		SHOOT
FLASH_ORANGE	1
IR



















































END_EVENT

EVENT HIT
MOTOR 30
SND	HURT
INC	nb_hit
LED_ON		30	1















































END_EVENT

EVENT ANIM_FINISHED
HUD_ICON_ON	BULLET
HUD_JAUGE_BLINK	1
END_EVENT

END_STATE
